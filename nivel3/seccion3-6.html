def hay__vocal_mayuscula(cadena: str) -> bool:
  hay_mayuscula = False
  i = 0
  while i < len(cadena) and not hay_mayuscula:
    letra_i = cadena[i]
    if letra_i in "AEIOU":
      hay_mayuscula = True
    else:
      i += 1
  
  return hay_mayuscula
